25static const unsigned char base64_table[65] =
26 "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789+/";
27static const unsigned char base64_table_last = '=';
28
29/**
30 * base64_encode - Base64 encode
31 * @src: Data to be encoded
32 * @len: Length of the data to be encoded
33 * @out_len: Pointer to output length variable, or %NULL if not used
34 * Returns: Allocated buffer of out_len bytes of encoded data,
35 * or %NULL on failure
36 *
37 * Caller is responsible for freeing the returned buffer. Returned buffer is
38 * nul terminated to make it easier to use as a C string. The nul terminator is
39 * not included in out_len.
40 */
41static inline unsigned char * base64_encode(const unsigned char *src, size_t len,
42 size_t *out_len)
43{
44 unsigned char *out, *pos;
45 const unsigned char *end, *in;
46 size_t olen;
47 int line_len;
48
49 olen = len * 4 / 3 + 4; /* 3-byte blocks to 4-byte */
50 olen += olen / 72; /* line feeds */
51 olen++; /* nul termination */
52 if (olen < len)
53 return NULL; /* integer overflow */
54 out = (unsigned char *)malloc(olen);
55 if (out == NULL)
56 return NULL;
57
58 end = src + len;
59 in = src;
60 pos = out;
61 line_len = 0;
62 while (end - in >= 3) {
63 *pos++ = base64_table[in[0] >> 2];
64 *pos++ = base64_table[((in[0] & 0x03) << 4) | (in[1] >> 4)];
65 *pos++ = base64_table[((in[1] & 0x0f) << 2) | (in[2] >> 6)];
66 *pos++ = base64_table[in[2] & 0x3f];
67 in += 3;
68 line_len += 4;
69 if (line_len >= 72) {
70 *pos++ = '\n';
71 line_len = 0;
72 }
73 }
74
75 if (end - in) {
76 *pos++ = base64_table[in[0] >> 2];
77 if (end - in == 1) {
78 *pos++ = base64_table[(in[0] & 0x03) << 4];
79 *pos++ = base64_table_last;
80 } else {
81 *pos++ = base64_table[((in[0] & 0x03) << 4) |
82 (in[1] >> 4)];
83 *pos++ = base64_table[(in[1] & 0x0f) << 2];
84 }
85 *pos++ = base64_table_last;
86 line_len += 4;
87 }
88
89 if (line_len)
90 *pos++ = '\n';
91
92 *pos = '\0';
93 if (out_len)
94 *out_len = (size_t)(pos - out);
95 return out;
96}

99/**
100 * base64_decode - Base64 decode
101 * @src: Data to be decoded
102 * @len: Length of the data to be decoded
103 * @out_len: Pointer to output length variable
104 * Returns: Allocated buffer of out_len bytes of decoded data,
105 * or %NULL on failure
106 *
107 * Caller is responsible for freeing the returned buffer.
108 */
109static inline unsigned char * base64_decode(const unsigned char *src, size_t len,
110 size_t *out_len)
111{
112 unsigned char dtable[256], *out, *pos, block[4], tmp;
113 size_t i, count, olen;
114 int pad = 0;
115
116
117 memset(dtable, 0x80, 256);
118 for (i = 0; i < sizeof(base64_table) - 1; i++)
119 dtable[base64_table[i]] = (unsigned char) i;
120 dtable[base64_table_last] = 0;
121
122 count = 0;
123 for (i = 0; i < len; i++) {
124 if (dtable[src[i]] != 0x80)
125 count++;
126 }
127
128 if (count == 0 || count % 4)
129 return NULL;
130
131 olen = count / 4 * 3;
132 pos = out = (unsigned char *)malloc(olen);
133 if (out == NULL)
134 return NULL;
135
136 count = 0;
137 for (i = 0; i < len; i++) {
138 tmp = dtable[src[i]];
139 if (tmp == 0x80)
140 continue;
141
142 if (src[i] == base64_table_last)
143 pad++;
144 block[count] = tmp;
145 count++;
146 if (count == 4) {
147 *pos++ = (unsigned char)((block[0] << 2) | (block[1] >> 4));
148 *pos++ = (unsigned char)((block[1] << 4) | (block[2] >> 2));
149 *pos++ = (unsigned char)((block[2] << 6) | block[3]);
150 count = 0;
151 if (pad) {
152 if (pad == 1)
153 pos--;
154 else if (pad == 2)
155 pos -= 2;
156 else {
157 /* Invalid padding */
158 free(out);
159 return NULL;
160 }
161 break;
162 }
163 }
164 }
165
166 *out_len = (size_t)(pos - out);
167 return out;
168}

582 //! \brief Write the bytes in the buffer to the stream as base64 encoded characters.
583 void WriteNext() {
584 if (buffer_pos_ == 0) return;
585 // Encode
586 unsigned char encoded[4] = {'\0', '\0', '\0', '\0'};
587 unsigned char *in = &(buffer_[0]);
588 unsigned char *pos = &(encoded[0]);
589 if ((!buffer_empty_[0]) && (!buffer_empty_[1]) && (!buffer_empty_[2])) {
590 *pos++ = base64_table[in[0] >> 2];
591 *pos++ = base64_table[((in[0] & 0x03) << 4) | (in[1] >> 4)];
592 *pos++ = base64_table[((in[1] & 0x0f) << 2) | (in[2] >> 6)];
593 *pos++ = base64_table[in[2] & 0x3f];
594 } else {
595 *pos++ = base64_table[in[0] >> 2];
596 if (buffer_empty_[1]) {
597 *pos++ = base64_table[(in[0] & 0x03) << 4];
598 *pos++ = base64_table_last;
599 } else {
600 *pos++ = base64_table[((in[0] & 0x03) << 4) |
601 (in[1] >> 4)];
602 *pos++ = base64_table[(in[1] & 0x0f) << 2];
603 }
604 *pos++ = base64_table_last;
605 }
606 // Output

608 if (encoded[i] == '\0')
609 break;
610 stream_.Put((Ch)(encoded[i]));
611 }
612 // Reset
613 buffer_pos_ = 0;
614 buffer_[0] = '\0';
615 buffer_[1] = '\0';
616 buffer_[2] = '\0';
617 buffer_empty_[0] = true;
618 buffer_empty_[1] = true;
619 buffer_empty_[2] = true;
620 }

650template<typename Encoding, typename BufferType>
651bool parseYggdrasilString(const typename Encoding::Ch* str, SizeType length, bool copy,
652 BufferType& os_body, BufferType& os_schema) {
653 if (!isYggdrasilString(str, length, copy))
654 return false;
655 const typename Encoding::Ch ygg[5] = {'-', 'Y', 'G', 'G', '-'};
656 SizeType len_ygg = 5;
657 // Locate -ygg- markers
658 SizeType i = len_ygg, beg_schema = len_ygg, end_schema, len_schema, elen_schema, beg_body, end_body, len_body, elen_body;
659 beg_schema = len_ygg;
660 while ((i < length) && (memcmp(ygg, str + i, sizeof(ygg)) != 0)) i++;
661 end_schema = i;
662 len_schema = end_schema - beg_schema;
663 elen_schema = len_schema * 3 / 4;

665 end_body = length - len_ygg;

667 elen_body = len_body * 3 / 4;

673 // Extract the schema

677 for (SizeType j = 0; j < elen_schema; j++) {
678 // std::cerr << " char " << j << " " << is64.Peek() << std::endl;
679 if (is64.Peek() != '\0') {
680 os_schema.Put(is64.Take());
681 } else {

683 }
684 }
685 is.src_ += len_ygg;
686 // Extract the body
687 // std::cerr << "Reading body (encoded_len = " << len_body
688 // <<", decoded_len = " << elen_body
689 // << ")" << std::endl;

693 if (!(is64.PeekEmpty())) {
694 os_body.Put(is64.Take());
695 } else {
696 is64.Take();
697 nempty_body++;
698 }
699 }
700 elen_body = elen_body - nempty_body;
701 is.src_ += len_ygg;
702 RAPIDJSON_ASSERT(is.Tell() == (size_t)length);
703 // std::cerr << "schema: \"" << os_schema.GetString() << "\"" << std::endl;
704 // std::cerr << "body: \"" << os_body.GetString() << "\"" << std::endl;
