62inline
63void Ply::fromObjWavefront(const ObjWavefront& o) {
64 size_t nvert = 0;
65 for (std::vector<ObjElement*>::const_iterator it = o.elements.begin(); it != o.elements.end(); it++) {
66 if ((*it)->code == "v") {
67 this->add_element("vertex", (*it)->get_double_array());
68 nvert++;
69 } else if ((*it)->code == "f") {
70 this->add_element("face", (*it)->get_int_array(nvert));
71 } else if ((*it)->code == "l") {
72 this->add_element("edge", (*it)->get_int_array(nvert));
73 }
74 else
75 RAPIDJSON_ASSERT(((*it)->code == "v") ||
76 ((*it)->code == "f") ||
77 ((*it)->code == "l"));
78 }

107#define ENCODING_STRING_(name, ...) \
108 const Ch name ## String[] = { __VA_ARGS__, '\0' }
109#define COMPARE_(x, dec) \
110 (std::basic_string<Ch>(x) == std::basic_string<Ch>(dec ## String))
111#define SWITCH_(x, CASE_) \
112 CASE_(x, ASCII, ASCII) \
113 else CASE_(x, ASCII, Null) \
114 else CASE_(x, UTF8, UTF8) \
115 else CASE_(x, UTF16, UTF16) \
116 else CASE_(x, UTF32, UTF32) \
117 else CASE_(x, UTF32, UCS4) \
118 else { \
119 return false; \
120 }
121template <typename SourceEncoding, typename DestEncoding, typename AllocatorType>
122inline bool TranslateEncoding_inner(const void* src, SizeType srcNbytes,
123 void*& dst, SizeType& dstNbytes,
124 AllocatorType& allocator,
125 bool requireFixedWidth = false) {
126 if (requireFixedWidth && (!DestEncoding::fixedWidth || !SourceEncoding::fixedWidth))
127 return false;
128 SizeType srcLength = srcNbytes / ((SizeType)sizeof(typename SourceEncoding::Ch));
129 GenericStringStream<SourceEncoding> is((const typename SourceEncoding::Ch*)src);
130 GenericStringBuffer<DestEncoding> os;
131 while (RAPIDJSON_LIKELY(is.Tell() < srcLength)) {
132 Transcoder<SourceEncoding, DestEncoding>::Transcode(is, os);
133 }
134 if (dst == NULL) {
135 dstNbytes = (SizeType)os.GetLength() * (SizeType)sizeof(typename DestEncoding::Ch);
136 dst = allocator.Malloc(dstNbytes);
137 if (!dst) return false;
138 } else {
139 if (dstNbytes != (SizeType)os.GetLength() * (SizeType)sizeof(typename DestEncoding::Ch))
140 return false;
141 }
142 memcpy(dst, os.GetString(), dstNbytes);
143 return true;
144}
145template <typename SourceEncoding, typename Ch, typename AllocatorType>
146inline bool TranslateEncoding_outer(const void* src, SizeType srcNbytes,
147 void*& dst, SizeType& dstNbytes, const Ch* dstEncoding,
148 AllocatorType& allocator,
149 bool requireFixedWidth = false) {
150 ENCODING_STRING_(ASCII, 'A', 'S', 'C', 'I', 'I');
151 ENCODING_STRING_(UTF8, 'U', 'T', 'F', '8');
152 ENCODING_STRING_(UTF16, 'U', 'T', 'F', '1', '6');
153 ENCODING_STRING_(UTF32, 'U', 'T', 'F', '3', '2');
154 ENCODING_STRING_(UCS4, 'U', 'C', 'S', '4');
155 ENCODING_STRING_(Null, 'n', 'u', 'l', 'l');
156#define INNER_CASE_(x, dec, decS) \
157 if (COMPARE_(x, decS)) { \
158 return TranslateEncoding_inner<SourceEncoding, dec<> >(src, srcNbytes, dst, dstNbytes, allocator, requireFixedWidth); \
159 }
160 SWITCH_(dstEncoding, INNER_CASE_)
161}
