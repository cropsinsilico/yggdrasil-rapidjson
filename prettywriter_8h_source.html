43//! Combination of PrettyWriter format flags.
44/*! \see PrettyWriter::SetFormatOptions
45 */
46enum PrettyFormatOptions {
47 kFormatDefault = 0, //!< Default pretty formatting.
48 kFormatSingleLineArray = 1 //!< Format arrays on a single line.
49};

51//! Writer with indentation and spacing.
52/*!
53 \tparam OutputStream Type of output os.
54 \tparam SourceEncoding Encoding of source string.
55 \tparam TargetEncoding Encoding of output stream.
56 \tparam StackAllocator Type of allocator for allocating memory of stack.
57*/
58template<typename OutputStream, typename SourceEncoding = UTF8<>, typename TargetEncoding = UTF8<>, typename StackAllocator = CrtAllocator, unsigned writeFlags = kWriteDefaultFlags>
59class PrettyWriter : public Writer<OutputStream, SourceEncoding, TargetEncoding, StackAllocator, writeFlags> {
60public:
61 typedef Writer<OutputStream, SourceEncoding, TargetEncoding, StackAllocator, writeFlags> Base;
62 typedef typename Base::Ch Ch;
63
64 //! Constructor
65 /*! \param os Output stream.
66 \param allocator User supplied allocator. If it is null, it will create a private one.
67 \param levelDepth Initial capacity of stack.
68 */
69 explicit PrettyWriter(OutputStream& os, StackAllocator* allocator = 0, size_t levelDepth = Base::kDefaultLevelDepth) :
70 Base(os, allocator, levelDepth), indentChar_(' '), indentCharCount_(4), formatOptions_(kFormatDefault) {}
71
72
73 explicit PrettyWriter(StackAllocator* allocator = 0, size_t levelDepth = Base::kDefaultLevelDepth) :
74 Base(allocator, levelDepth), indentChar_(' '), indentCharCount_(4), formatOptions_(kFormatDefault) {}
75
76#if RAPIDJSON_HAS_CXX11_RVALUE_REFS
77 PrettyWriter(PrettyWriter&& rhs) :
78 Base(std::forward<PrettyWriter>(rhs)), indentChar_(rhs.indentChar_), indentCharCount_(rhs.indentCharCount_), formatOptions_(rhs.formatOptions_) {}
79#endif

81 //! Set custom indentation.
82 /*! \param indentChar Character for indentation. Must be whitespace character (' ', '\\t', '\\n', '\\r').
83 \param indentCharCount Number of indent characters for each indentation level.
84 \note The default indentation is 4 spaces.
85 */
86 PrettyWriter& SetIndent(Ch indentChar, unsigned indentCharCount) {
87 RAPIDJSON_ASSERT(indentChar == ' ' || indentChar == '\t' || indentChar == '\n' || indentChar == '\r');
88 indentChar_ = indentChar;
89 indentCharCount_ = indentCharCount;
90 return *this;
91 }
92
93 //! Set pretty writer formatting options.
94 /*! \param options Formatting options.
95 */
96 PrettyWriter& SetFormatOptions(PrettyFormatOptions options) {
97 formatOptions_ = options;
98 return *this;
99 }

106#ifdef RAPIDJSON_YGGDRASIL
107#define RAPIDJSON_WRAP_BASE64_(method, arg1) \
108 if (this->w64p_) return this->w64_()->method arg1

113 bool Null() { RAPIDJSON_WRAP_BASE64_(Null, ()); PrettyPrefix(kNullType); return Base::EndValue(Base::WriteNull()); }
114 bool Bool(bool b) { RAPIDJSON_WRAP_BASE64_(Bool, (b)); PrettyPrefix(b ? kTrueType : kFalseType); return Base::EndValue(Base::WriteBool(b)); }
115 bool Int(int i) { RAPIDJSON_WRAP_BASE64_(Int, (i)); PrettyPrefix(kNumberType); return Base::EndValue(Base::WriteInt(i)); }
116 bool Uint(unsigned u) { RAPIDJSON_WRAP_BASE64_(Uint, (u)); PrettyPrefix(kNumberType); return Base::EndValue(Base::WriteUint(u)); }
117 bool Int64(int64_t i64) { RAPIDJSON_WRAP_BASE64_(Int64, (i64)); PrettyPrefix(kNumberType); return Base::EndValue(Base::WriteInt64(i64)); }
118 bool Uint64(uint64_t u64) { RAPIDJSON_WRAP_BASE64_(Uint64, (u64)); PrettyPrefix(kNumberType); return Base::EndValue(Base::WriteUint64(u64)); }

121 bool RawNumber(const Ch* str, SizeType length, bool copy = false) {
122 RAPIDJSON_ASSERT(str != 0);
123 (void)copy;
124 PrettyPrefix(kNumberType);
125 return Base::EndValue(Base::WriteString(str, length));

129 RAPIDJSON_WRAP_BASE64_(String, (str, length, copy));

131 (void)copy;
132 PrettyPrefix(kStringType);

142 bool StartObject() {
143 RAPIDJSON_WRAP_BASE64_(StartObject, ());
144 PrettyPrefix(kObjectType);
145 new (Base::level_stack_.template Push<typename Base::Level>()) typename Base::Level(false);
146 return Base::WriteStartObject();
147 }
148
149 bool Key(const Ch* str, SizeType length, bool copy = false) { return String(str, length, copy); }
150
151#if RAPIDJSON_HAS_STDSTRING
152 bool Key(const std::basic_string<Ch>& str) {
153 return Key(str.data(), SizeType(str.size()));
154 }
155#endif
156
157 bool EndObject(SizeType memberCount = 0) {
158 RAPIDJSON_WRAP_BASE64_(EndObject, (memberCount));
159 (void)memberCount;
160 RAPIDJSON_ASSERT(Base::level_stack_.GetSize() >= sizeof(typename Base::Level)); // not inside an Object
161 RAPIDJSON_ASSERT(!Base::level_stack_.template Top<typename Base::Level>()->inArray); // currently inside an Array, not Object
162 RAPIDJSON_ASSERT(0 == Base::level_stack_.template Top<typename Base::Level>()->valueCount % 2); // Object has a Key without a Value
163
164 bool empty = Base::level_stack_.template Pop<typename Base::Level>(1)->valueCount == 0;
165
166 if (!empty) {
167 Base::os_->Put('\n');
168 WriteIndent();
169 }
170 bool ret = Base::EndValue(Base::WriteEndObject());
171 (void)ret;
172 RAPIDJSON_ASSERT(ret == true);
173 if (Base::level_stack_.Empty()) // end of json text
174 Base::Flush();
175 return true;
176 }
177
178 bool StartArray() {
179 RAPIDJSON_WRAP_BASE64_(StartArray, ());
180 PrettyPrefix(kArrayType);
181 new (Base::level_stack_.template Push<typename Base::Level>()) typename Base::Level(true);
182 return Base::WriteStartArray();
183 }
184
185 bool EndArray(SizeType memberCount = 0) {
186 RAPIDJSON_WRAP_BASE64_(EndArray, (memberCount));
187 (void)memberCount;
188 RAPIDJSON_ASSERT(Base::level_stack_.GetSize() >= sizeof(typename Base::Level));
189 RAPIDJSON_ASSERT(Base::level_stack_.template Top<typename Base::Level>()->inArray);
190 bool empty = Base::level_stack_.template Pop<typename Base::Level>(1)->valueCount == 0;
191
192 if (!empty && !(formatOptions_ & kFormatSingleLineArray)) {
193 Base::os_->Put('\n');
194 WriteIndent();
195 }
196 bool ret = Base::EndValue(Base::WriteEndArray());
197 (void)ret;
198 RAPIDJSON_ASSERT(ret == true);
199 if (Base::level_stack_.Empty()) // end of json text
200 Base::Flush();
201 return true;
202 }

216 //! Write a raw JSON value.
217 /*!
218 For user to write a stringified JSON as a value.
219
220 \param json A well-formed JSON value. It should not contain null character within [0, length - 1] range.
221 \param length Length of the json.
222 \param type Type of the root of json.
223 \note When using PrettyWriter::RawValue(), the result json may not be indented correctly.
224 */
225 bool RawValue(const Ch* json, size_t length, Type type) {
226 RAPIDJSON_ASSERT(json != 0);
227 PrettyPrefix(type);
228 return Base::EndValue(Base::WriteRawValue(json, length));
229 }

231#undef RAPIDJSON_WRAP_BASE64_
232
233protected:
234#ifdef RAPIDJSON_YGGDRASIL
235 void Prefix(Type type) OVERRIDE_CXX11 { PrettyPrefix(type); }
236#endif // RAPIDJSON_YGGDRASIL
237 void PrettyPrefix(Type type) {
238 (void)type;
239 if (Base::level_stack_.GetSize() != 0) { // this value is not at root
240 typename Base::Level* level = Base::level_stack_.template Top<typename Base::Level>();
241
242 if (level->inArray) {
243 if (level->valueCount > 0) {
244 Base::os_->Put(','); // add comma if it is not the first element in array
245 if (formatOptions_ & kFormatSingleLineArray)
246 Base::os_->Put(' ');
247 }
248
249 if (!(formatOptions_ & kFormatSingleLineArray)) {
250 Base::os_->Put('\n');
251 WriteIndent();
252 }
253 }
254 else { // in object
255 if (level->valueCount > 0) {
256 if (level->valueCount % 2 == 0) {
257 Base::os_->Put(',');
258 Base::os_->Put('\n');
259 }
260 else {
261 Base::os_->Put(':');
262 Base::os_->Put(' ');
263 }
264 }
265 else
266 Base::os_->Put('\n');
267
268 if (level->valueCount % 2 == 0)
269 WriteIndent();
270 }
271 if (!level->inArray && level->valueCount % 2 == 0)
272 RAPIDJSON_ASSERT(type == kStringType); // if it's in object, then even number should be a name
273 level->valueCount++;
274 }
275 else {
276 RAPIDJSON_ASSERT(!Base::hasRoot_); // Should only has one and only one root.
277 Base::hasRoot_ = true;
278 }
279 }
280
281 void WriteIndent() {
282 size_t count = (Base::level_stack_.GetSize() / sizeof(typename Base::Level)) * indentCharCount_;
283 PutN(*Base::os_, static_cast<typename OutputStream::Ch>(indentChar_), count);
284 }
285
286 Ch indentChar_;
287 unsigned indentCharCount_;
288 PrettyFormatOptions formatOptions_;
289
290private:
291 // Prohibit copy constructor & assignment operator.
292 PrettyWriter(const PrettyWriter&);
293 PrettyWriter& operator=(const PrettyWriter&);
294};
